172def autocorr_penalty(returns):
173    """Metric to account for auto correlation"""
174    if isinstance(returns, _pd.DataFrame):
175        returns = returns[returns.columns[0]]
176
177    # returns.to_csv('/Users/ran/Desktop/test.csv')
178    num = len(returns)
179    coef = _np.abs(_np.corrcoef(returns[:-1], returns[1:])[0, 1])
180    corr = [((num - x) / num) * coef**x for x in range(1, num)]
181    return _np.sqrt(1 + 2 * _np.sum(corr))

187def sharpe(returns, rf=0.0, periods=252, annualize=True, smart=False):
188    """
189    Calculates the sharpe ratio of access returns
190
191    If rf is non-zero, you must specify periods.
192    In this case, rf is assumed to be expressed in yearly (annualized) terms
193
194    Args:
195        * returns (Series, DataFrame): Input return series
196        * rf (float): Risk-free rate expressed as a yearly (annualized) return
197        * periods (int): Freq. of returns (252/365 for daily, 12 for monthly)
198        * annualize: return annualize sharpe?
199        * smart: return smart sharpe ratio
200    """
201    if rf != 0 and periods is None:
202        raise Exception("Must provide periods if rf != 0")
203
204    returns = _utils._prepare_returns(returns, rf, periods)
205    divisor = returns.std(ddof=1)
206    if smart:
207        # penalize sharpe with auto correlation
208        divisor = divisor * autocorr_penalty(returns)
209    res = returns.mean() / divisor
210
211    if annualize:
212        return res * _np.sqrt(1 if periods is None else periods)
213
214    return res

221def sortino(returns, rf=0, periods=252, annualize=True, smart=False):
222    """
223    Calculates the sortino ratio of access returns
224
225    If rf is non-zero, you must specify periods.
226    In this case, rf is assumed to be expressed in yearly (annualized) terms
227
228    Calculation is based on this paper by Red Rock Capital
229    http://www.redrockcapital.com/Sortino__A__Sharper__Ratio_Red_Rock_Capital.pdf
230    """
231    if rf != 0 and periods is None:
232        raise Exception("Must provide periods if rf != 0")
233
234    returns = _utils._prepare_returns(returns, rf, periods)
235
236    downside = _np.sqrt((returns[returns < 0] ** 2).sum() / len(returns))
237
238    if smart:
239        # penalize sortino with auto correlation
240        downside = downside * autocorr_penalty(returns)
241
242    res = returns.mean() / downside
243
244    if annualize:
245        return res * _np.sqrt(1 if periods is None else periods)
246
247    return res

254def rolling_sortino(
255    returns, rf=0, rolling_period=126, annualize=True, periods_per_year=252, **kwargs
256):
257    if rf != 0 and rolling_period is None:
258        raise Exception("Must provide periods if rf != 0")
259
260    if kwargs.get("prepare_returns", True):
261        returns = _utils._prepare_returns(returns, rf, rolling_period)
262
263    downside = (
264        returns.rolling(rolling_period).apply(
265            lambda x: (x.values[x.values < 0] ** 2).sum()
266        )
267        / rolling_period
268    )
269
270    res = returns.rolling(rolling_period).mean() / _np.sqrt(downside)
271    if annualize:
272        res = res * _np.sqrt(1 if periods_per_year is None else periods_per_year)
273    return res

276def adjusted_sortino(returns, rf=0, periods=252, annualize=True, smart=False):
277    """
278    Jack Schwager's version of the Sortino ratio allows for
279    direct comparisons to the Sharpe. See here for more info:
280    https://archive.is/wip/2rwFW
281    """
282    data = sortino(returns, rf, periods=periods, annualize=annualize, smart=smart)
283    return data / _sqrt(2)

286def probabilistic_ratio(
287    series, rf=0.0, base="sharpe", periods=252, annualize=False, smart=False
288):
289    if base.lower() == "sharpe":
290        base = sharpe(series, periods=periods, annualize=False, smart=smart)
291    elif base.lower() == "sortino":
292        base = sortino(series, periods=periods, annualize=False, smart=smart)
293    elif base.lower() == "adjusted_sortino":
294        base = adjusted_sortino(series, periods=periods, annualize=False, smart=smart)
295    else:
296        raise Exception(
297            "`metric` must be either `sharpe`, `sortino`, or `adjusted_sortino`"
298        )
299    skew_no = skew(series)
300    kurtosis_no = kurtosis(series)
301
302    n = len(series)
303
304    sigma_sr = _np.sqrt(
305        (
306            1
307            + (0.5 * base**2)
308            - (skew_no * base)
309            + (((kurtosis_no - 3) / 4) * base**2)
310        )
311        / (n - 1)
312    )
313
314    ratio = (base - rf) / sigma_sr
315    psr = _norm.cdf(ratio)
316
317    if annualize:
318        return psr * (252**0.5)
319    return psr

330def omega(returns, required_return=0.0, periods=252):
331    """
332    Determines the Omega ratio of a strategy.
333    See https://en.wikipedia.org/wiki/Omega_ratio for more details.
334    """
335    if len(returns) < 2:
336        return _np.nan
337
338    if required_return <= -1:
339        return _np.nan
340
341    if periods == 1:
342        return_threshold = required_return
343    else:
344        return_threshold = (1 + required_return) ** (1.0 / periods) - 1
345
346    returns_less_thresh = returns - return_threshold
347    numer = returns_less_thresh[returns_less_thresh > 0.0].sum()
348    denom = -1.0 * returns_less_thresh[returns_less_thresh < 0.0].sum()
349
350    if denom > 0.0:
351        return numer / denom
352
353    return _np.nan

356def gain_to_pain_ratio(returns, rf=0, resolution="D"):
357    """
358    Jack Schwager's GPR. See here for more info:
359    https://archive.is/wip/2rwFW
360    """
361    returns = _utils._prepare_returns(returns, rf).resample(resolution).sum()
362    downside = abs(returns[returns < 0].sum())
363    return returns.sum() / downside

366def cagr(returns, rf=0.0, compounded=True, periods=252):
367    """
368    Calculates the communicative annualized growth return
369    (CAGR%) of access returns
370
371    If rf is non-zero, you must specify periods.
372    In this case, rf is assumed to be expressed in yearly (annualized) terms
373    """
374    total = _utils._prepare_returns(returns, rf)
375    if compounded:
376        total = comp(total)
377    else:
378        total = _np.sum(total)
379
380    years = (returns.index[-1] - returns.index[0]).days / periods
381
382    res = abs(total + 1.0) ** (1.0 / years) - 1
383
384    if isinstance(returns, _pd.DataFrame):
385        res = _pd.Series(res)
386        res.index = returns.columns
387
388    return res

464def value_at_risk(returns, sigma=1, confidence=0.95):
465    """
466    Calculats the daily value-at-risk
467    (variance-covariance calculation with confidence n)
468    """
469    mu = returns.mean()
470    sigma *= returns.std()
471
472    if confidence > 1:
473        confidence = confidence / 100
474
475    return _norm.ppf(1 - confidence, mu, sigma)

583def drawdown_details(drawdown):
584    """
585    Calculates drawdown details, including start/end/valley dates,
586    duration, max drawdown and max dd for 99% of the dd period
587    for every drawdown period
588    """
589
590    def _drawdown_details(drawdown):
591        # mark no drawdown
592        no_dd = drawdown == 0
593
594        # extract dd start dates, first date of the drawdown
595        starts = ~no_dd & no_dd.shift(1)
596        starts = list(starts[starts.values].index)
597
598        # extract end dates, last date of the drawdown
599        ends = no_dd & (~no_dd).shift(1)
600        ends = ends.shift(-1, fill_value=False)
601        ends = list(ends[ends.values].index)
602
603        # no drawdown :)
604        if not starts:
605            return _pd.DataFrame(
606                index=[],
607                columns=(
608                    "start",
609                    "valley",
610                    "end",
611                    "days",
612                    "max drawdown",
613                    "99% max drawdown",
614                ),
615            )
616
617        # drawdown series begins in a drawdown
618        if ends and starts[0] > ends[0]:
619            starts.insert(0, drawdown.index[0])
620
621        # series ends in a drawdown fill with last date
622        if not ends or starts[-1] > ends[-1]:
623            ends.append(drawdown.index[-1])
624
625        # build dataframe from results
626        data = []
627        for i, _ in enumerate(starts):
628            dd = drawdown[starts[i] : ends[i]]
629            clean_dd = -remove_outliers(-dd, 0.99)
630            data.append(
631                (
632                    starts[i],
633                    dd.idxmin(),
634                    ends[i],
635                    (ends[i] - starts[i]).days + 1,
636                    dd.min() * 100,
637                    clean_dd.min() * 100,
638                )
639            )
640
641        df = _pd.DataFrame(
642            data=data,
643            columns=(
644                "start",
645                "valley",
646                "end",
647                "days",
648                "max drawdown",
649                "99% max drawdown",
650            ),
651        )
652        df["days"] = df["days"].astype(int)
653        df["max drawdown"] = df["max drawdown"].astype(float)
654        df["99% max drawdown"] = df["99% max drawdown"].astype(float)
655
656        df["start"] = df["start"].dt.strftime("%Y-%m-%d")
657        df["end"] = df["end"].dt.strftime("%Y-%m-%d")
658        df["valley"] = df["valley"].dt.strftime("%Y-%m-%d")
659
660        return df
661
662    if isinstance(drawdown, _pd.DataFrame):
663        _dfs = {}
664        for col in drawdown.columns:
665            _dfs[col] = _drawdown_details(drawdown[col])
666        return _pd.concat(_dfs, axis=1)
667
668    return _drawdown_details(drawdown)

671def kelly_criterion(returns, prepare_returns=True):
672    """
673    Calculates the recommended maximum amount of capital that
674    should be allocated to the given strategy, based on the
675    Kelly Criterion (http://en.wikipedia.org/wiki/Kelly_criterion)
676    """
677    if prepare_returns:
678        returns = _utils._prepare_returns(returns)
679    win_loss_ratio = payoff_ratio(returns)
680    win_prob = win_rate(returns)
681    lose_prob = 1 - win_prob

596        starts = ~no_dd & no_dd.shift(1)
597        starts = list(starts[starts.values].index)
598
599        # extract end dates, last date of the drawdown
600        ends = no_dd & (~no_dd).shift(1)
601        ends = ends.shift(-1, fill_value=False)
602        ends = list(ends[ends.values].index)
603
604        # no drawdown :)
605        if not starts:
606            return _pd.DataFrame(
607                index=[],
608                columns=(
609                    "start",
610                    "valley",
611                    "end",
612                    "days",
613                    "max drawdown",
614                    "99% max drawdown",
615                ),
616            )
617
618        # drawdown series begins in a drawdown
619        if ends and starts[0] > ends[0]:
620            starts.insert(0, drawdown.index[0])
621
622        # series ends in a drawdown fill with last date
623        if not ends or starts[-1] > ends[-1]:
624            ends.append(drawdown.index[-1])
625
626        # build dataframe from results
627        data = []
628        for i, _ in enumerate(starts):
629            dd = drawdown[starts[i] : ends[i]]
630            clean_dd = -remove_outliers(-dd, 0.99)
631            data.append(
632                (
633                    starts[i],
634                    dd.idxmin(),
635                    ends[i],
636                    (ends[i] - starts[i]).days + 1,
637                    dd.min() * 100,
638                    clean_dd.min() * 100,
639                )
640            )
641
642        df = _pd.DataFrame(
643            data=data,
644            columns=(
645                "start",
646                "valley",
647                "end",
648                "days",
649                "max drawdown",
650                "99% max drawdown",
651            ),
652        )
653        df["days"] = df["days"].astype(int)
654        df["max drawdown"] = df["max drawdown"].astype(float)
655        df["99% max drawdown"] = df["99% max drawdown"].astype(float)
656
657        df["start"] = df["start"].dt.strftime("%Y-%m-%d")
658        df["end"] = df["end"].dt.strftime("%Y-%m-%d")
659        df["valley"] = df["valley"].dt.strftime("%Y-%m-%d")
660
661        return df
662
663    if isinstance(drawdown, _pd.DataFrame):
664        _dfs = {}
665        for col in drawdown.columns:
666            _dfs[col] = _drawdown_details(drawdown[col])
667        return _pd.concat(_dfs, axis=1)
668
669    return _drawdown_details(drawdown)
