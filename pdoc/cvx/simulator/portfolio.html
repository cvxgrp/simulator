cvx.simulator.portfolio

Portfolio representation and analysis for the CVX Simulator.

This module provides the Portfolio class, which represents a portfolio of assets with methods for calculating various metrics (NAV, profit, drawdown, etc.) and analyzing performance. The Portfolio class is typically created by the Builder class after a simulation is complete.

 34@dataclass(frozen=True)
 35class Portfolio:
 36    """
 37    Represents a portfolio of assets with methods for analysis and visualization.
 38
 39    The Portfolio class is a frozen dataclass (immutable) that represents a portfolio
 40    of assets with their prices and positions (units). It provides methods for
 41    calculating various metrics like NAV, profit, drawdown, and for visualizing
 42    the portfolio's performance.
 43
 44    Attributes
 45    ----------
 46    prices : pd.DataFrame
 47        DataFrame of asset prices over time, with dates as index and assets as columns
 48    units : pd.DataFrame
 49        DataFrame of asset positions (units) over time, with dates as index and assets as columns
 50    aum : Union[float, pd.Series]
 51        Assets under management, either as a constant float or as a Series over time
 52    """
 53
 54    prices: pd.DataFrame
 55    units: pd.DataFrame
 56    aum: float | pd.Series
 57
 58    def __post_init__(self) -> None:
 59        """
 60        Validate the portfolio data after initialization.
 61
 62        This method is automatically called after an instance of the Portfolio
 63        class has been initialized. It performs a series of validation checks
 64        to ensure that the prices and units dataframes are in the expected format
 65        with no duplicates or missing data.
 66
 67        The method checks that:
 68        - Both prices and units dataframes have monotonic increasing indices
 69        - Both prices and units dataframes have unique indices
 70        - The index of units is a subset of the index of prices
 71        - The columns of units is a subset of the columns of prices
 72
 73        Raises
 74        ------
 75        AssertionError
 76            If any of the validation checks fail
 77        """
 78
 79        assert self.prices.index.is_monotonic_increasing
 80        assert self.prices.index.is_unique
 81        assert self.units.index.is_monotonic_increasing
 82        assert self.units.index.is_unique
 83
 84        assert set(self.units.index).issubset(set(self.prices.index))
 85        assert set(self.units.columns).issubset(set(self.prices.columns))
 86
 87    @property
 88    def index(self) -> pd.DatetimeIndex:
 89        """
 90        Get the time index of the portfolio.
 91
 92        Returns
 93        -------
 94        pd.DatetimeIndex
 95            A DatetimeIndex representing the time period for which portfolio
 96            data is available
 97
 98        Notes
 99        -----
100        This property extracts the index from the prices DataFrame, which
101        represents all time points in the portfolio history.
102        """
103        return pd.DatetimeIndex(self.prices.index)
104
105    @property
106    def assets(self) -> pd.Index:
107        """
108        Get the list of assets in the portfolio.
109
110        Returns
111        -------
112        pd.Index
113            An Index containing the names of all assets in the portfolio
114
115        Notes
116        -----
117        This property extracts the column names from the prices DataFrame,
118        which correspond to all assets for which price data is available.
119        """
120        return self.prices.columns
121
122    @property
123    def nav(self) -> pd.Series:
124        """
125        Get the net asset value (NAV) of the portfolio over time.
126
127        The NAV represents the total value of the portfolio at each point in time.
128        If aum is provided as a Series, it is used directly. Otherwise, the NAV
129        is calculated from the cumulative profit plus the initial aum.
130
131        Returns
132        -------
133        pd.Series
134            Series representing the NAV of the portfolio over time
135        """
136        if isinstance(self.aum, pd.Series):
137            series = self.aum
138        else:
139            profit = (self.cashposition.shift(1) * self.returns.fillna(0.0)).sum(axis=1)
140            series = profit.cumsum() + self.aum
141
142        series.name = "NAV"
143        return series
144
145    @property
146    def profit(self) -> pd.Series:
147        """
148        Get the profit/loss of the portfolio at each time point.
149
150        This calculates the profit or loss at each time point based on the
151        previous positions and the returns of each asset.
152
153        Returns
154        -------
155        pd.Series
156            Series representing the profit/loss at each time point
157
158        Notes
159        -----
160        The profit is calculated by multiplying the previous day's positions
161        (in currency terms) by the returns of each asset, and then summing
162        across all assets.
163        """
164        series = (self.cashposition.shift(1) * self.returns.fillna(0.0)).sum(axis=1)
165        series.name = "Profit"
166        return series
167
168    @property
169    def highwater(self) -> pd.Series:
170        """
171        Get the high-water mark of the portfolio over time.
172
173        The high-water mark represents the highest value the portfolio has
174        reached up to each point in time.
175
176        Returns
177        -------
178        pd.Series
179            Series representing the high-water mark at each time point
180
181        Notes
182        -----
183        This is calculated using an expanding maximum of the NAV, which means
184        at each point in time, it shows the maximum NAV achieved up to that point.
185        This is commonly used to calculate drawdowns and performance fees.
186        """
187        series = self.nav.expanding(min_periods=1).max()
188        series.name = "Highwater"
189        return series
190
191    @property
192    def drawdown(self) -> pd.Series:
193        """
194        Get the drawdown of the portfolio over time.
195
196        Drawdown measures the decline in portfolio value from its previous
197        highest point (high-water mark) to the current point, expressed as
198        a fraction of the high-water mark.
199
200        Returns
201        -------
202        pd.Series
203            Series representing the drawdown at each time point
204
205        Notes
206        -----
207        Calculated as 1 - (NAV / high-water mark). A positive drawdown means
208        the portfolio is currently worth less than its high-water mark.
209        For example, a drawdown of 0.1 means the NAV is currently 90% of
210        the high-water mark.
211        """
212        series = 1.0 - self.nav / self.highwater
213        series.name = "Drawdown"
214        return series
215
216    @property
217    def cashposition(self) -> pd.DataFrame:
218        """
219        Get the cash value of each position over time.
220
221        This calculates the cash value of each position by multiplying
222        the number of units by the price for each asset at each time point.
223
224        Returns
225        -------
226        pd.DataFrame
227            DataFrame with the cash value of each position over time,
228            with dates as index and assets as columns
229        """
230        return self.prices * self.units
231
232    @property
233    def returns(self) -> pd.DataFrame:
234        """
235        Get the returns of individual assets over time.
236
237        This calculates the percentage change in price for each asset
238        from one time point to the next.
239
240        Returns
241        -------
242        pd.DataFrame
243            DataFrame with the returns of each asset over time,
244            with dates as index and assets as columns
245        """
246        return self.prices.pct_change()
247
248    @property
249    def trades_units(self) -> pd.DataFrame:
250        """
251        Get the trades made in the portfolio in terms of units.
252
253        This calculates the changes in position (units) from one time point
254        to the next for each asset.
255
256        Returns
257        -------
258        pd.DataFrame
259            DataFrame with the trades (changes in units) for each asset over time,
260            with dates as index and assets as columns
261
262        Notes
263        -----
264        Calculated as the difference between consecutive position values.
265        Positive values represent buys, negative values represent sells.
266        The first row contains the initial positions, as there are no previous
267        positions to compare with.
268        """
269        t = self.units.fillna(0.0).diff()
270        t.loc[self.index[0]] = self.units.loc[self.index[0]]
271        return t.fillna(0.0)
272
273    @property
274    def trades_currency(self) -> pd.DataFrame:
275        """
276        Get the trades made in the portfolio in terms of currency.
277
278        This calculates the cash value of trades by multiplying the changes
279        in position (units) by the current prices.
280
281        Returns
282        -------
283        pd.DataFrame
284            DataFrame with the cash value of trades for each asset over time,
285            with dates as index and assets as columns
286
287        Notes
288        -----
289        Calculated by multiplying trades_units by prices.
290        Positive values represent buys (cash outflows),
291        negative values represent sells (cash inflows).
292        """
293        return self.trades_units * self.prices
294
295    @property
296    def turnover_relative(self) -> pd.DataFrame:
297        """
298        Get the turnover relative to the portfolio NAV.
299
300        This calculates the trades as a percentage of the portfolio NAV,
301        which provides a measure of trading activity relative to portfolio size.
302
303        Returns
304        -------
305        pd.DataFrame
306            DataFrame with the relative turnover for each asset over time,
307            with dates as index and assets as columns
308
309        Notes
310        -----
311        Calculated by dividing trades_currency by NAV.
312        Positive values represent buys, negative values represent sells.
313        A value of 0.05 means a buy equal to 5% of the portfolio NAV.
314        """
315        return self.trades_currency.div(self.nav, axis=0)
316
317    @property
318    def turnover(self) -> pd.DataFrame:
319        """
320        Get the absolute turnover in the portfolio.
321
322        This calculates the absolute value of trades in currency terms,
323        which provides a measure of total trading activity regardless of
324        direction (buy or sell).
325
326        Returns
327        -------
328        pd.DataFrame
329            DataFrame with the absolute turnover for each asset over time,
330            with dates as index and assets as columns
331
332        Notes
333        -----
334        Calculated as the absolute value of trades_currency.
335        This is useful for calculating trading costs that apply equally
336        to buys and sells.
337        """
338        return self.trades_currency.abs()
339
340    def __getitem__(self, time: datetime | str | pd.Timestamp) -> pd.Series:
341        """
342        Get the portfolio positions (units) at a specific time.
343
344        This method allows for dictionary-like access to the portfolio positions
345        at a specific time point using the syntax: portfolio[time].
346
347        Parameters
348        ----------
349        time : Union[datetime, str, pd.Timestamp]
350            The time index for which to retrieve the positions
351
352        Returns
353        -------
354        pd.Series
355            Series containing the positions (units) for each asset at the specified time
356
357        Raises
358        ------
359        KeyError
360            If the specified time is not in the portfolio's index
361
362        Examples
363        --------
364        >>> portfolio['2023-01-01']  # Get positions on January 1, 2023
365        >>> portfolio[pd.Timestamp('2023-01-01')]  # Same as above
366        """
367        return self.units.loc[time]
368
369    @property
370    def equity(self) -> pd.DataFrame:
371        """
372        Get the equity (cash value) of each position over time.
373
374        This property returns the cash value of each position in the portfolio,
375        calculated by multiplying the number of units by the price for each asset.
376
377        Returns
378        -------
379        pd.DataFrame
380            DataFrame with the cash value of each position over time,
381            with dates as index and assets as columns
382
383        Notes
384        -----
385        This is an alias for the cashposition property and returns the same values.
386        The term "equity" is used in the context of the cash value of positions,
387        not to be confused with the equity asset class.
388        """
389        return self.cashposition
390
391    @property
392    def weights(self) -> pd.DataFrame:
393        """
394        Get the weight of each asset in the portfolio over time.
395
396        This calculates the relative weight of each asset in the portfolio
397        by dividing the cash value of each position by the total portfolio
398        value (NAV) at each time point.
399
400        Returns
401        -------
402        pd.DataFrame
403            DataFrame with the weight of each asset over time,
404            with dates as index and assets as columns
405
406        Notes
407        -----
408        The sum of weights across all assets at any given time should equal 1.0
409        for a fully invested portfolio with no leverage. Weights can be negative
410        for short positions.
411        """
412        return self.equity.apply(lambda x: x / self.nav)
413
414    @property
415    def data(self):
416        frame = self.nav.pct_change().to_frame()
417        frame.index.name = "Date"
418        return build_data(returns=frame)
419
420    @classmethod
421    def from_cashpos_prices(cls, prices: pd.DataFrame, cashposition: pd.DataFrame, aum: float):
422        """Build Futures Portfolio from cashposition"""
423        units = cashposition.div(prices, fill_value=0.0)
424        return cls(prices=prices, units=units, aum=aum)
425
426    @classmethod
427    def from_cashpos_returns(cls, returns: pd.DataFrame, cashposition: pd.DataFrame, aum: float):
428        """Build Futures Portfolio from cashposition"""
429        prices = returns2prices(returns)
430        return cls.from_cashpos_prices(prices, cashposition, aum)
431
432    def snapshot(self, title: str = "Portfolio Summary", log_scale: bool = True):
433        return self.data.plots.plot_snapshot(title=title, log_scale=log_scale)

124    def nav(self) -> pd.Series:
125        """
126        Get the net asset value (NAV) of the portfolio over time.
127
128        The NAV represents the total value of the portfolio at each point in time.
129        If aum is provided as a Series, it is used directly. Otherwise, the NAV
130        is calculated from the cumulative profit plus the initial aum.
131
132        Returns
133        -------
134        pd.Series
135            Series representing the NAV of the portfolio over time
136        """
137        if isinstance(self.aum, pd.Series):
138            series = self.aum
139        else:
140            profit = (self.cashposition.shift(1) * self.returns.fillna(0.0)).sum(axis=1)
141            series = profit.cumsum() + self.aum
142
143        series.name = "NAV"
144        return series
